&НаСервере
Процедура ЗаполнитьИзФайлаНаСервере(АдресДанных)

	ФайлПриемник = ПолучитьИмяВременногоФайла("xlsx");
	
	ДанныеХранилища = ПолучитьИзВременногоХранилища(АдресДанных);
	ДанныеХранилища.Записать(ФайлПриемник);
	ТабДок = Новый ТабличныйДокумент;
	ТабДок.Прочитать(ФайлПриемник);   
	
	ПЗ = Новый ПостроительЗапроса;
	ПЗ.ИсточникДанных = Новый ОписаниеИсточникаДанных(ТабДок.Область(2,1,ТабДок.ВысотаТаблицы,11));//передаем ТЗ
	ПЗ.ДобавлениеПредставлений = ТипДобавленияПредставлений.НеДобавлять;
	ПЗ.ЗаполнитьНастройки();
	ПЗ.Выполнить();
	Выборка = ПЗ.Результат.Выбрать();
	
	Объект.Результаты.Очистить();
	
	Пока Выборка.Следующий() Цикл
		//ОбработкаПрерыванияПользователя();
		
		Если Не ЗначениеЗаполнено(СокрЛП(Выборка.Участник)) Тогда
			Продолжить;
		КонецЕсли;
		
		Участник = Справочники.Спортсмены.НайтиПоНаименованию(Выборка.Участник);
		Если НЕ ЗначениеЗаполнено(Участник) И СокрЛП(Выборка.Участник) <> "" Тогда
			//создадим нового спортсмена
			ФИО = СокрЛП(Выборка.Участник);
			Фамилия = "";
			Имя  = "";
			Отчество = "";
			ПозицияПробел = СтрНайти(ФИО, " ");
			Если ПозицияПробел <> 0 Тогда
				Фамилия = Лев(ФИО, ПозицияПробел - 1);
				ФИО = СокрЛП(Сред(ФИО, ПозицияПробел)); 
				ПозицияПробел = СтрНайти(ФИО, " "); 
				Если ПозицияПробел <> 0 Тогда
					Имя = Лев(ФИО, ПозицияПробел - 1);
					Отчество = СокрЛП(Сред(ФИО, ПозицияПробел));
				Иначе
					Имя = ФИО;	
				КонецЕсли;
			Иначе
				Фамилия = ФИО;	
			КонецЕсли;
			НовыйУчастник = Справочники.Спортсмены.СоздатьЭлемент();
			НовыйУчастник.Наименование = Выборка.Участник;
			НовыйУчастник.Фамилия = Фамилия;
			НовыйУчастник.Имя = Имя;
			НовыйУчастник.Отчество = Отчество;
			НовыйУчастник.Записать();
			Сообщить("Добавлен новый спортсмен " + Выборка.Участник);
			Участник = НовыйУчастник.Ссылка;
		КонецЕсли;		
		
		Дисциплина = Справочники.СпортивныеДисциплины.НайтиПоНаименованию(Выборка.Дисциплина);
		
		НС = Объект.Результаты.Добавить();
		НС.Участник	= Участник;
		НС.Дисциплина = Дисциплина;
		НС.Результат = Выборка.Результат;
		Попытка
			НС.Место = Число(Выборка.Место);
		Исключение
		КонецПопытки;
		
	КонецЦикла;
	
КонецПроцедуры
