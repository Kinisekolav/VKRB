Процедура ОтправитьНапоминаниеНаСервере()

		|	ДвоичныеДанныеФайлов.Файл.Файл КАК Файл,

	КонецЕсли;

	ОтправитьНапоминаниеНаСервере();
	
КонецПроцедуры

Процедура ОтправитьНаПочту(Получатели, ТаблицаФайлов, Соревнование) Экспорт
	
	УчетнаяЗаписьЭлектроннойПочты = Справочники.УчетныеЗаписиЭлектроннойПочты.СистемнаяУчетнаяЗаписьЭлектроннойПочты;
	Если УчетнаяЗаписьЭлектроннойПочты = Справочники.УчетныеЗаписиЭлектроннойПочты.ПустаяСсылка() И УчетнаяЗаписьЭлектроннойПочты = Неопределено тогда
		Текст_Ошибки = ВернутьСДатой("ОШИБКА! Не определена учетная запись электронной почты!");
		Сообщить(Текст_Ошибки);
		Возврат;
	КонецЕсли;
	
	СтруктураПараметровПодключения = новый Структура;
	СтруктураПараметровПодключения.Вставить("АдресСервераSMTP", УчетнаяЗаписьЭлектроннойПочты.СерверИсходящейПочты);
	СтруктураПараметровПодключения.Вставить("АдресСервераPOP3", УчетнаяЗаписьЭлектроннойПочты.СерверВходящейПочты);
	СтруктураПараметровПодключения.Вставить("ПортPOP3", УчетнаяЗаписьЭлектроннойПочты.ПортСервераВходящейПочты);
	СтруктураПараметровПодключения.Вставить("ПортSMTP", УчетнаяЗаписьЭлектроннойПочты.ПортСервераИсходящейПочты);
	СтруктураПараметровПодключения.Вставить("ПользовательSMTP", УчетнаяЗаписьЭлектроннойПочты.ПользовательSMTP);
	СтруктураПараметровПодключения.Вставить("ПарольSMTP", УчетнаяЗаписьЭлектроннойПочты.Пароль);
	СтруктураПараметровПодключения.Вставить("Пользователь", УчетнаяЗаписьЭлектроннойПочты.Пользователь);
	СтруктураПараметровПодключения.Вставить("Пароль", УчетнаяЗаписьЭлектроннойПочты.Пароль);
	СтруктураПараметровПодключения.Вставить("АутентификацияSMTP", СпособSMTPАутентификации.ПоУмолчанию);
	
	ТекстПисьма = "Добрый день!" + Символы.ПС +
		"Вы зарегистрированы в соревновании """ + Соревнование.Название + """" + Символы.ПС + 
	    "Вид спорта: " + Соревнование.ВидСпорта + Символы.ПС +
		"Место проведения: " + Соревнование.МестоПроведения + Символы.ПС +
		"Дата: " + ?(Соревнование.ДатаНачала = Соревнование.ДатаОкончания, Формат(Соревнование.ДатаНачала, "ДЛФ=DD"),
			Формат(Соревнование.ДатаНачала, "ДЛФ=DD") + " - " + Формат(Соревнование.ДатаОкончания, "ДЛФ=DD"));
	
	СтруктПарам = Новый Структура;
	СтруктПарам.Вставить("Тема", "Участие в соревновании");
	СтруктПарам.Вставить("ТекстСообщения", ТекстПисьма);
	СтруктПарам.Вставить("ИмяОтправителя", УчетнаяЗаписьЭлектроннойПочты.ИмяПользователя);

	ОтправительПочтовогоСообщения = УчетнаяЗаписьЭлектроннойПочты.Пользователь;
	
	Профиль = Новый ИнтернетПочтовыйПрофиль;
	ЗаполнитьЗначенияСвойств(Профиль, СтруктураПараметровПодключения);
	
	Сообщение = Новый ИнтернетПочтовоеСообщение;
	Сообщение.Тема = СтруктПарам.Тема;
	Сообщение.Отправитель = ОтправительПочтовогоСообщения;
	Сообщение.ИмяОтправителя = СтруктПарам.ИмяОтправителя;

	ТекстСообщения = Сообщение.Тексты.Добавить(СтруктПарам.ТекстСообщения);
	ТекстСообщения.ТипТекста = ТипТекстаПочтовогоСообщения.ПростойТекст;
	
	Если Получатели.Количество() = 0 тогда 
		Текст_Ошибки = ВернутьСДатой("ОШИБКА! Нет получателей для отправки файлов!");
		Сообщить(Текст_Ошибки);
	КонецЕсли;

	Для Каждого Получатель Из Получатели Цикл
		Сообщение.Получатели.Добавить(СокрЛП(Получатель));	
	КонецЦикла;
	
	Для Каждого ТекФайл Из ТаблицаФайлов Цикл
		Вложение = ТекФайл.Файл.Получить();
	    Сообщение.Вложения.Добавить(Вложение, ТекФайл.ИмяФайла);	
	КонецЦикла;  
	
Профиль.АдресСервераPOP3 = "";
Профиль.ПортPOP3 = 0;
Профиль.ПользовательSMTP = Профиль.Пользователь;
//Профиль.ПарольIMAP = Профиль.ПарольSMTP
	
	Почта = Новый ИнтернетПочта;
	
	Попытка
		Почта.Подключиться(Профиль);
		Почта.Послать(Сообщение);  
		Сообщить("Сообщение успешно отправлено!");
	Исключение
		Сообщить(ВернутьСДатой("ОШИБКА! Не удалось отправить! " + Символы.ПС + ОписаниеОшибки()));
		Сообщить(ОписаниеОшибки(), СтатусСообщения.Важное);
	КонецПопытки;
	
	Почта.Отключиться();
	
КонецПроцедуры
