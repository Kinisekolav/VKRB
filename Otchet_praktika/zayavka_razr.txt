&НаКлиенте
Процедура СпортсменыДокументРезультатовСоревнованийПриИзменении(Элемент)
	
	ТекДанные = Элементы.Спортсмены.ТекущиеДанные;
	Если ТекДанные = Неопределено Тогда
		Возврат;
	КонецЕсли;
	
	СтруктураРезультатовСоревнований = ПолучитьСтруктуруРезультатовСоревнований(ТекДанные.ДокументРезультатовСоревнований, ТекДанные.Спортсмен, Объект.Разряд);
	ТекДанные.НазваниеСоревнований = СтруктураРезультатовСоревнований.НазваниеСоревнований;
	ТекДанные.Результат = СтруктураРезультатовСоревнований.Результат;
		
КонецПроцедуры

&НаСервере
Функция ПолучитьСтруктуруРезультатовСоревнований(РезультатСоревнований, Спортсмен, Разряд)
	
	Запрос = Новый Запрос;
	Запрос.Текст = 
		"ВЫБРАТЬ ПЕРВЫЕ 1
		|	РезультатыСоревнованийРезультаты.Результат КАК Результат,
		|	РезультатыСоревнованийРезультаты.ВыполненныйРазряд КАК ВыполненныйРазряд
		|ИЗ
		|	Документ.РезультатыСоревнований.Результаты КАК РезультатыСоревнованийРезультаты
		|ГДЕ
		|	РезультатыСоревнованийРезультаты.Участник = &Участник
		|	И РезультатыСоревнованийРезультаты.Ссылка = &Ссылка
		|	И РезультатыСоревнованийРезультаты.ВыполненныйРазряд = &Разряд";
	
	Запрос.УстановитьПараметр("Ссылка", РезультатСоревнований);
	Запрос.УстановитьПараметр("Участник", Спортсмен);
	Запрос.УстановитьПараметр("Разряд", Разряд);
	
	РезультатЗапроса = Запрос.Выполнить();
	
	ВыборкаДетальныеЗаписи = РезультатЗапроса.Выбрать();
	
	Рез = Новый Структура("НазваниеСоревнований, Результат");
	Рез.НазваниеСоревнований = РезультатСоревнований.Соревнование.Название;
	Если ВыборкаДетальныеЗаписи.Следующий() Тогда
		Рез.Результат  = ВыборкаДетальныеЗаписи.Результат;
	КонецЕсли;
		
	Возврат Рез;	
	
КонецФункции
